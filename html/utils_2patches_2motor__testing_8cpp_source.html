56void writeMotor(int L, int R) {
57 /* L: Left Motor analog output
58 * R: Right Motor analog output
59 * range: -255 to 255
60 */
61
62 if (L >= 0) {
63 lw_Forward();
64 } else {
65 lw_Backward();
66 }
67
68 if (R >= 0) {
69 rw_Forward();
70 } else {
71 rw_Backward();
72 }
73 gpioPWM(13, abs(L));
74 gpioPWM(12, abs(R));
75 cout << L << R << "\n";
76 return;
77}
